// BlobSort

function swap(arr, i, j) {
  var temp = arr[i];
  arr[i] = arr[j];
  arr[j] = temp;
}

function blobSort(arr) {
  for (var i = 0; i < arr.length; i++) {
    for (var j = 0; j < arr.length; j++) {
      if (arr[j] > arr[i]) {
        swap(arr, i, j);
      }
    }
  }
}

var arr = [5, 3, 0, -1, 9];
blobSort(arr);
